% Supply Projection Graph
% Shows circulating supply growth over 20 years
%
% ACCESSIBILITY ALT TEXT:
% A line graph showing total BTH supply over 20 years. X-axis shows years,
% Y-axis shows total supply in millions of BTH. The curve rises steeply
% during the 5 annual halvings, reaching the Phase-1 supply of ~611M BTH at
% year 5, then continues growing at 2% net annual inflation (compounding,
% supply-dependent) through the tail phase, reaching ~885M by year 20. Botho's
% supply grows indefinitely at a fixed 2% net rate, unlike Bitcoin's 21M cap.

\begin{figure}[ht]
\centering
% y plotted in units of 100M BTH (data / 100) to keep coordinates small.
\begin{tikzpicture}[xscale=0.5, yscale=4]
    % Axes (y in units of 100M BTH; labels show M BTH)
    \draw[->] (0,0) -- (21,0) node[right] {\small Years};
    \draw[->] (0,0) -- (0,9.8) node[above] {\small Supply (M BTH)};

    % Y-axis labels (value 2 = 200M, etc.)
    \foreach \y/\lbl in {0/0,2/200,4/400,6/600,8/800} {
        \draw (-0.3,\y) -- (0.3,\y) node[left=3pt] {\tiny \lbl};
    }

    % X-axis labels
    \foreach \x in {0,5,10,15,20} {
        \draw (\x,-0.2) -- (\x,0.2) node[below=3pt] {\tiny \x};
    }

    % Grid
    \foreach \y in {2,4,6,8} {
        \draw[gray!30, dashed] (0,\y) -- (20,\y);
    }

    % Botho supply curve: annual halvings then 2% net (2.5% gross) tail
    % Y1=315.4, Y2=473.0, Y3=551.9, Y4=591.3, Y5=611.0 (Phase-1),
    % Y6=626.3, Y10=691.3, Y15=782, Y20=884.9  (values / 100)
    \draw[blue, very thick]
        (0,0) -- (1,3.154) -- (2,4.730) -- (3,5.519)
        -- (4,5.913) -- (5,6.110)
        .. controls (8,6.50) and (12,7.20) .. (15,7.82)
        .. controls (17,8.25) and (19,8.65) .. (20,8.849);

    % Bitcoin comparison (21M cap, scaled x10 for visibility -> 210M)
    \draw[red, thick, dashed]
        (0,0)
        .. controls (2,1.20) and (4,1.70) .. (4,1.80)
        .. controls (8,2.00) and (12,2.05) .. (16,2.10)
        -- (20,2.10);

    % Key milestones
    \node[anchor=west, font=\scriptsize, fill=white] at (1.3,3.30) {Year 1: 315M};
    \node[anchor=south, font=\scriptsize, fill=white] at (5,6.15) {Year 5: 611M (Phase-1)};
    \node[anchor=south, font=\scriptsize, fill=white] at (12,7.20) {Year 10: 691M};
    \node[anchor=east, font=\scriptsize, fill=white] at (20,8.849) {Year 20: 885M};

    % Legend
    \draw[blue, very thick] (1,9.2) -- (3,9.2) node[right, font=\scriptsize] {Botho};
    \draw[red, thick, dashed] (1,8.6) -- (3,8.6) node[right, font=\scriptsize] {Bitcoin (scaled $\times$10)};

    % Tail emission annotation
    \draw[<->, gray, thick] (5,1.20) -- (20,1.20) node[midway, below, font=\scriptsize] {Tail: 2\% net/yr (compounding)};
\end{tikzpicture}
\caption{Projected circulating supply over 20 years. The five annual halvings
(50, 25, 12.5, 6.25, 3.125 BTH/block) fix the Phase-1 supply at
$\approx$611M BTH by year 5; from year 6 onward the supply grows at a fixed
2\% net annual rate (2.5\% gross before the expected fee burn), reaching
$\approx$885M by year 20. Unlike Bitcoin's hard cap at 21M, Botho's supply
grows indefinitely at a fixed percentage rate, ensuring permanent security
funding. (Bitcoin curve scaled $\times$10 for visibility.)}
\label{fig:supply-projection}
\end{figure}
