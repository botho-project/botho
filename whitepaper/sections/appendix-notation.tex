\section{Notation Reference}
\label{app:notation}

This appendix provides a comprehensive reference for notation used
throughout the paper.

\subsection{Mathematical Notation}

\begin{longtable}{@{}cl@{}}
\toprule
\textbf{Symbol} & \textbf{Description} \\
\midrule
\endhead

\multicolumn{2}{l}{\textit{Groups and Fields}} \\
$\G$ & Elliptic curve group (Ristretto255) \\
$G$ & Generator of $\G$ \\
$H$ & Secondary generator for Pedersen commitments \\
$q$ & Order of $\G$ (prime) \\
$\Zq$ & Integers modulo $q$ \\
$\Rq$ & Polynomial ring for lattice operations \\
\\

\multicolumn{2}{l}{\textit{Keys}} \\
$a, b$ & View and spend private keys \\
$A, B$ & View and spend public keys \\
$(C_i, D_i)$ & Subaddress public key pair for index $i$ \\
$P$ & One-time public key \\
$x$ & One-time private key \\
$I$ & Key image \\
\\

\multicolumn{2}{l}{\textit{Hash Functions}} \\
$\Hs(\cdot)$ & Hash function to scalar \\
$\Hp(\cdot)$ & Hash function to curve point \\
$\Hash(\cdot)$ & General cryptographic hash \\
\\

\multicolumn{2}{l}{\textit{Ring Signatures}} \\
$\{P_i\}_{i=0}^{n-1}$ & Ring of public keys \\
$\pi$ & Secret index of real key in ring \\
$\sigma$ & Signature \\
$n$ & Ring size (default: 20) \\
\\

\multicolumn{2}{l}{\textit{Commitments}} \\
$C$ & Pedersen commitment \\
$C_v$ & Commitment to value $v$ \\
$r$ & Blinding factor \\
\\

\multicolumn{2}{l}{\textit{Operations}} \\
$\xleftarrow{\$}$ & Uniform random sampling \\
$\|$ & Concatenation \\
$\oplus$ & Bitwise XOR \\
$\mod q$ & Modular reduction \\
\\

\multicolumn{2}{l}{\textit{Complexity}} \\
$\negl(\lambda)$ & Negligible function in $\lambda$ \\
$\poly(\lambda)$ & Polynomial function in $\lambda$ \\
$\lambda$ & Security parameter \\
\\

\bottomrule
\end{longtable}
